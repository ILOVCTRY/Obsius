## 分帧

```python
import os
from PIL import Image
def get_path():
    print("输入目标路径：如 D:/Dowload/45663022307c456897d30639f56da759/out.gif")
    path=input()
    print(path)
    path=path.replace("\\","/")
    path=path.replace('"',"")
    print(path)
    return path

def gif_split(target_gif_path):#拆分gif
    img=Image.open(target_gif_path)
    try:#使用这语句再img.seek无法移动下一帧时循环会停止
        while True:
            save_path="C:/Users/Nan/Desktop/out/"
            current=img.tell()#获取当前处理的帧的索引
            img.save(save_path+str(current+1)+'.png')
            img.seek(current+1)#切换到下一帧
    except:
        pass
    print("OK")

gif_split(get_path())


```

## 分帧后合并一张图片

- 从左到右合并
>

  >montage flag*.png -tile x1 -geometry +0+0 flag.png     #合并图片
  >```
  >
  >>参数解释
  >>
  >>- -tile x1 里，由于 x 前面没有数字，所以每行的图像数量没有限制；而 x 后面是 1，这表明所有图像会按照一列的方式进行排列。
  >>- +0+0 代表图像之间在水平和垂直方向上都没有间距。如果是 +5+5，就表示图像之间水平和垂直方向上的间距都是 5 个像素。



## identify工具

```
identify -format "%T" flag.gif
```

> 参数介绍
>
> - **`"%T"`**：格式说明符，`%T`表示获取图像的文件类型，双引号是为了在命令行中正确传递包含特殊字符（这里`%`是特殊字符 ）的参数。
> - **`%w`**：获取图像宽度。比如`identify -format "%w" flag.gif` ，会输出`flag.gif`图像的宽度像素值。
> - **`%h`**：获取图像高度。使用`identify -format "%h" flag.gif` ，可得到图像高度像素值。
> - **`%m`**：获取图像的色彩模式，如`RGB`、`CMYK`等 。执行`identify -format "%m" flag.gif` ，会输出对应的色彩模式。

import os

def get_path():

def get_color_of_images(imae_path):#判断一个文件夹内的图片颜色转化为二进制
    ret=""
    line=""
    for i in range(0,24):
        for j in range(0,24):
            x = j * 10 + 5
            y = i * 10 + 5
            image_path=f"{imae_path}/{i*24+j+1}.png"
            #print(image_path)
            img=Image.open(image_path)
            img=img.convert("RGB")
            img_array=img.load()
            r,g,b=img_array[x,y]
            if g == 255:
                line += "0"
                print(line)
            if r == 255 and b == 255:
                line += "1"
            if len(line) == 8:
                print(line)
                ret += chr(int(line, 2))
                line = ""
    print(f'结果是{ret}')
    print(len(ret))

get_color_of_images(get_path())

## 修改宽高

- 随波逐流工具

- 010
  
- crc爆破宽高

  ```pythn
  import zlib
  import struct
  import argparse
  import itertools
   
   
  parser = argparse.ArgumentParser()
  parser.add_argument("-f", type=str, default=None, required=True,
                      help="输入同级目录下图片的名称")
  args  = parser.parse_args()
   
   
  bin_data = open(args.f, 'rb').read()
  crc32key = zlib.crc32(bin_data[12:29]) # 计算crc
  original_crc32 = int(bin_data[29:33].hex(), 16) # 原始crc
   
   
  if crc32key == original_crc32: # 计算crc对比原始crc
      print('宽高没有问题!')
  else:
      input_ = input("宽高被改了, 是否CRC爆破宽高? (Y/n):")
      if input_ not in ["Y", "y", ""]:
          exit()
      else: 
          for i, j in itertools.product(range(4095), range(4095)): # 理论上0x FF FF FF FF，但考虑到屏幕实际/cpu，0x 0F FF就差不多了，也就是4095宽度和高度
              data = bin_data[12:16] + struct.pack('>i', i) + struct.pack('>i', j) + bin_data[24:29]
              crc32 = zlib.crc32(data)
              if(crc32 == original_crc32): # 计算当图片大小为i:j时的CRC校验值，与图片中的CRC比较，当相同，则图片大小已经确定
                  print(f"\nCRC32: {hex(original_crc32)}")
                  print(f"宽度: {i}, hex: {hex(i)}")
                  print(f"高度: {j}, hex: {hex(j)}")
                  exit(0)
  ```
